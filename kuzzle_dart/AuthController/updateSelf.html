Future<User> updateSelf(Map<String, dynamic> body) async {
  final response = await kuzzle.query(KuzzleRequest(
    controller: name,
    action: 'updateSelf',
    body: body,
  ));

  return User.fromKuzzleResponse(kuzzle, response);
}
